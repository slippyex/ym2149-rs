Core Chip Emulation

Direct register access, cycle-accurate emulation

Raw YM2149 Access

Use ym2149-core for direct chip control:

use ym2149::{Ym2149, Ym2149Backend};

fn main() {
    let mut chip = Ym2149::new();

    // Set channel A frequency (440 Hz)
    chip.write_register(0, 0x1C);  // Tone A Lo
    chip.write_register(1, 0x01);  // Tone A Hi
    chip.write_register(8, 0x0F);  // Volume A (max)
    chip.write_register(7, 0x3E);  // Mixer: enable tone A

    // Generate samples
    for _ in 0..44100 {
        chip.clock();
        let sample = chip.get_sample();
        // Send sample to audio output...
    }
}

Register Map

R0-1 Tone A
R2-3 Tone B
R4-5 Tone C
R6 Noise
R7 Mixer
R8-10 Volume
R11-12 Env Period
R13 Env Shape

Standalone Playback

Play YM files without a game engine

Load and Play YM Files

Use ym2149-ym-replayer for YM2-YM6 formats:

use ym2149_ym_replayer::{load_song, PlaybackController};

fn main() -> Result<(), Box<dyn Error>> {
    // Load YM file
    let data = std::fs::read("music.ym")?;
    let (mut player, summary) = load_song(&data)?;

    println!("Title: {}", summary.title);
    println!("Author: {}", summary.author);
    println!("Duration: {}s", summary.duration_seconds);

    // Start playback
    player.play()?;

    // Generate samples (e.g., for cpal audio output)
    let samples = player.generate_samples(44100);
    Ok(())
}

With cpal Audio Output

Real-time playback using the cpal audio library:

use cpal::traits::{DeviceTrait, HostTrait, StreamTrait};
use parking_lot::Mutex;
use std::sync::Arc;

let player = Arc::new(Mutex::new(player));
let player_clone = Arc::clone(&player);

let host = cpal::default_host();
let device = host.default_output_device().unwrap();
let config = device.default_output_config().unwrap();

let stream = device.build_output_stream(
    &config.into(),
    move |data: &mut [f32], _| {
        let mut p = player_clone.lock();
        p.generate_samples_into(data);
    },
    |err| eprintln!("Audio error: {}", err),
    None,
).unwrap();

stream.play().unwrap();
std::thread::sleep(Duration::from_secs(60));

WebAssembly / Browser

Play chiptunes in the browser

JavaScript Integration

Use the pre-built WASM module:

import init, { Ym2149Player } from './ym2149_wasm.js';

async function playChiptune(fileData) {
    await init();

    // Create player from file data (Uint8Array)
    const player = new Ym2149Player(fileData);

    // Get metadata
    const meta = player.metadata;
    console.log(`Playing: ${meta.title} by ${meta.author}`);

    // Setup Web Audio API
    const ctx = new AudioContext({ sampleRate: 44100 });
    const processor = ctx.createScriptProcessor(4096, 1, 1);

    processor.onaudioprocess = (e) => {
        const output = e.outputBuffer.getChannelData(0);
        const samples = player.generateSamples(output.length);
        for (let i = 0; i < output.length; i++) {
            output[i] = samples[i];
        }
    };

    processor.connect(ctx.destination);
    player.play();
}

Player API

play() Start playback
pause() Pause playback
stop() Stop and reset
set_volume(v) 0.0 - 1.0
seek_to_percentage(p) 0.0 - 1.0
set_channel_mute(ch, b) Mute channel
next_subsong() AKS/SNDH
generateSamples(n) Get n samples

GIST Sound Effects

Atari ST synthesizer patches for game SFX

High-Level Player

Quick playback with GistPlayer:

use ym2149_gist_replayer::{GistPlayer, GistSound};

// Load a GIST .snd effect
let explosion = GistSound::load("explosion.snd")?;
let laser = GistSound::load("laser.snd")?;

// Create player (uses its own YM2149 chip)
let mut player = GistPlayer::new();

// Play sound on any free channel
player.play_sound(&explosion, None, None);

// Generate audio while playing
while player.is_playing() {
    let samples = player.generate_samples(882); // ~20ms
    // Send to audio output...
}

Dedicated SFX Chip (Music + Effects)

Use a second YM2149 for SFX alongside music:

// Main music player
let mut music_player = load_song(&ym_data)?;
music_player.play()?;

// Separate GIST player for sound effects
let mut sfx_player = GistPlayer::new();
let jump_sfx = GistSound::load("jump.snd")?;

// In your audio callback, mix both
fn generate_audio(output: &mut [f32]) {
    let music = music_player.generate_samples(output.len());
    let sfx = sfx_player.generate_samples(output.len());

    for i in 0..output.len() {
        output[i] = (music[i] + sfx[i]) * 0.5; // Mix 50/50
    }
}

// Trigger SFX on game events
if player_jumped {
    sfx_player.play_sound(&jump_sfx, None, None);
}

Low-Level Driver

Direct control with GistDriver:

use ym2149::Ym2149;
use ym2149_gist_replayer::{GistDriver, GistSound, TICK_RATE};

let sound = GistSound::load("effect.snd")?;
let mut chip = Ym2149::new();
let mut driver = GistDriver::new();

// Start sound (channel=None auto-selects, priority=i16::MAX-1)
driver.snd_on(&mut chip, &sound, None, None, -1, i16::MAX - 1);

// Call tick() at 200 Hz (GIST Timer C rate)
while driver.is_playing() {
    driver.tick(&mut chip);
    // Generate samples from chip...
}

Bevy Game Engine

Full integration for games and demos

Basic Setup

use bevy::prelude::*;
use bevy_ym2149::{Ym2149Playback, Ym2149Plugin};

fn main() {
    App::new()
        .add_plugins(DefaultPlugins)
        .add_plugins(Ym2149Plugin::default())
        .add_systems(Startup, setup)
        .run();
}

fn setup(mut commands: Commands) {
    commands.spawn(Camera2d);
    commands.spawn(Ym2149Playback::new("music.ym"));
}

Playlists with Crossfade

use bevy_ym2149::playlist::{
    CrossfadeConfig, PlaylistMode, PlaylistSource,
    Ym2149Playlist, Ym2149PlaylistPlayer,
};

fn setup(
    mut commands: Commands,
    mut playlists: ResMut<Assets<Ym2149Playlist>>,
) {
    let playlist = playlists.add(Ym2149Playlist {
        tracks: vec![
            PlaylistSource::Asset { path: "music/track1.ym".into() },
            PlaylistSource::Asset { path: "music/track2.aks".into() },
        ],
        mode: PlaylistMode::Loop,
    });

    commands.spawn((
        Ym2149Playback::default(),
        Ym2149PlaylistPlayer::with_crossfade(
            playlist,
            CrossfadeConfig::start_at_seconds(3.0),
        ),
    ));
}

commands.spawn((Ym2149Playback::new("demo.ym"), triggers));

use ym2149_common::ChiptunePlayer;

fn play_any<P: ChiptunePlayer>(mut player: P) {
    player.play();

    let meta = player.metadata();
    println!("Playing: {} by {}", meta.title, meta.author);

    let samples = player.generate_samples(4096);
    // All players produce f32 samples at configured sample rate
}

CLI Replayer

Terminal-based player with TUI visualization

# Install
cargo install ym2149-replayer-cli

# Play a single file
ym2149-replayer music.ym

# Play a directory (playlist mode)
ym2149-replayer ./music/

# Headless mode (no TUI)
ym2149-replayer --headless music.ym

# Export to WAV
ym2149-replayer --export output.wav music.ym
